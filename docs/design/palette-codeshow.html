/// Execute a single run lane and stream events back to the caller.
/// Aborts cleanly on cancellation; preserves partial trace on error.
#[derive(Debug, Clone)]
pub struct RunConfig<'cfg> {
    pub lane_id:    LaneId,
    pub timeout_ms: u64,     // 0 = no timeout
    pub policy:     &'cfg Policy,
    pub retries:    3,
}

#[instrument(skip(store, tx))]
pub async fn execute_lane<S: TraceStore>(
    cfg:   RunConfig<'_>,
    store: &mut S,
    tx:    mpsc::Sender<RunEvent>,
) -> Result<Trace, RunError> {
    let span = tracing::info_span!("lane.run", lane = %cfg.lane_id);
    let _g = span.enter();

    match store.claim_lane(cfg.lane_id).await {
        Ok(lease)  => run_with_lease(cfg, lease, tx).await,
        Err(e)     => Err(RunError::Claim(e))
    }
}

import React, { useState, useEffect } from "react";

interface RunSummaryProps<T> {
  laneId:   string;
  events:  T[];
  onRetry: (id: string) => void;
}

export function RunSummary<T>(
  { laneId, events, onRetry }: RunSummaryProps<T>
) {
  const [status, setStatus] = useState<Status>("idle");
  useEffect(() => {
    if (events.some(e => e.type === "failure"))
      setStatus("failed");
  }, [events]);
  return <LaneCard id={laneId} status={status} />;
}

from dataclasses import dataclass, field
from typing import AsyncGenerator, ClassVar

@dataclass
@retry(max_attempts=3, on=[ClaimError])
class AgentRunner:
    """Run a single agent lane and yield events."""
    lane_id:  str
    policy:   Policy
    retries:  int = 3

    async def run(self) -> AsyncGenerator[RunEvent, None]:
        async with self._claim() as lease:
            async for ev in self.policy.execute(lease):
                yield ev

Bash scripts/deploy.sh 
#!/usr/bin/env bash
set -euo pipefail

NAMESPACE="${NAMESPACE:-production}"
REGISTRY="registry.internal/anvil"
TAG="$(git rev-parse --short HEAD)"

log() { echo "[deploy] $*" >&2; }

if [[ "$1" == "--dry-run" ]]; then
  log "dry-run mode"
else
  docker build -t "${REGISTRY}:${TAG}" .
  kubectl -n "$NAMESPACE" set image \
    deploy/anvil-runner runner="${REGISTRY}:${TAG}"
fi
